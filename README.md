import turtle

t = turtle.Turtle()

for c in [
        'red', 'lime', 'violet', 'black', 'blue', 'yellow', 'grey', 'orange',
        'green', 'magenta', 'silver', 'gold', 'brown', 'pink', 'lightgrey',
        'cyan','dark green','cream'
]:
    t.color(c)
    t.forward(25)
    t.speed(8)
    t.left(20)
